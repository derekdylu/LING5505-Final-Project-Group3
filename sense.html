def print_word_pos_sentence (word_sentence, pos_sentence):

        print(f"{word}({pos})", end="\u3000")
    print()
    return

    cnt = 0

        snese_tree = word[i].senses
        cnt += len(word[i].senses)
        if(verbal == True): print(snese_tree)

    return cnt
